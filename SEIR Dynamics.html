function SEIR_simple_ode(du, u, p, t)
    S,E,I,R = u
    β,α,γ= p
    du[1] = -β*S*I
    du[2] = β*S*I - α*E
    du[3] = α*E - γ*I
    du[4] = γ*I
end

function SEIR_mortal_natal_ode(du, u, p, t)
    S,E,I,R = u
    β,α,γ,ν,μ= p
    du[1] = -β*S*I + ν*(S+E+I+R) - μ*S
    du[2] = β*S*I - α*E - μ*E
    du[3] = α*E - γ*I - μ*I
    du[4] = γ*I - μ*R
end
